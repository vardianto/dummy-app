68  @Override
69  public boolean onInterceptTouchEvent(MotionEvent ev) {
70  /*
71  * This method JUST determines whether we want to intercept the motion.
72  * If we return true, onMotionEvent will be called and we do the actual
73  * scrolling there.
74  */
75 
76  /*
77  * Shortcut the most recurring case: the user is in the dragging
78  * state and he is moving his finger. We want to intercept this
79  * motion.
80  */
81  final int action = ev.getAction();
82  if ((action == MotionEvent.ACTION_MOVE) && (mIsBeingDragged)) {
83  return true;
84  }
85 
86  if (!canScroll()) {
87  mIsBeingDragged = false;
88  return false;
89  }
90 
91  final float y = ev.getY();
92  final float x = ev.getX();
93 
94  switch (action) {
95  case MotionEvent.ACTION_MOVE:
96  /*
97  * mIsBeingDragged == false, otherwise the shortcut would have caught it. Check
98  * whether the user has moved far enough from his original down touch.
99  */
100 
101  /*
102  * Locally do absolute value. mLastMotionY is set to the y value
103  * of the down event.
104  */
105  final int yDiff = (int) Math.abs(y - mLastMotionY);
106  final int xDiff = (int) Math.abs(x - mLastMotionX);
107  if (yDiff > mTouchSlop || xDiff > mTouchSlop) {
108  mIsBeingDragged = true;
109  }
110  break;
111 
112  case MotionEvent.ACTION_DOWN:
113  /* Remember location of down touch */
114  mLastMotionY = y;
115  mLastMotionX = x;
116 
117  /*
118  * If being flinged and user touches the screen, initiate drag...
119  * otherwise don't. mScroller.isFinished should be false when
120  * being flinged.
121  */
122  mIsBeingDragged = !mScroller.isFinished();
123  break;
124 
125  case MotionEvent.ACTION_CANCEL:
126  case MotionEvent.ACTION_UP:
127  /* Release the drag */
128  mIsBeingDragged = false;
129  break;
130  }
131 
132  /*
133  * The only time we want to intercept motion events is if we are in the
134  * drag mode.
135  */
136  return mIsBeingDragged;
137  }

139  @Override
140  public boolean onTouchEvent(MotionEvent ev) {
141  if (ev.getAction() == MotionEvent.ACTION_DOWN && ev.getEdgeFlags() != 0) {
142  // Don't handle edge touches immediately -- they may actually belong to one of our descendants.
143  return false;
144  }
145 
146  if (!canScroll()) {
147  return false;
148  }
149 
150  final int action = ev.getAction();
151  final float y = ev.getY();
152  final float x = ev.getX();
153 
154  switch (action) {
155  case MotionEvent.ACTION_DOWN:
156  /*
157  * If being flinged and user touches, stop the fling. isFinished
158  * will be false if being flinged.
159  */
160  if (!mScroller.isFinished()) {
161  mScroller.abortAnimation();
162  }
163 
164  // Remember where the motion event started
165  mLastMotionY = y;
166  mLastMotionX = x;
167  break;
168  case MotionEvent.ACTION_MOVE:
169  // Scroll to follow the motion event
170  int deltaX = (int) (mLastMotionX - x);
171  int deltaY = (int) (mLastMotionY - y);
172  mLastMotionX = x;
173  mLastMotionY = y;
174 
175  if (deltaX < 0) {
176  if (getScrollX() < 0) {
177  deltaX = 0;
178  }
179  } else if (deltaX > 0) {
180  final int rightEdge = getWidth() - getPaddingRight();
181  final int availableToScroll = getChildAt(0).getRight() - getScrollX() - rightEdge;
182  if (availableToScroll > 0) {
183  deltaX = Math.min(availableToScroll, deltaX);
184  } else {
185  deltaX = 0;
186  }
187  }
188  if (deltaY < 0) {
189  if (getScrollY() < 0) {
190  deltaY = 0;
191  }
192  } else if (deltaY > 0) {
193  final int bottomEdge = getHeight() - getPaddingBottom();
194  final int availableToScroll = getChildAt(0).getBottom() - getScrollY() - bottomEdge;
195  if (availableToScroll > 0) {
196  deltaY = Math.min(availableToScroll, deltaY);
197  } else {
198  deltaY = 0;
199  }
200  }
201  if (deltaY != 0 || deltaX != 0)
202  scrollBy(deltaX, deltaY);
203  break;
204  }
205  return true;
206  }
